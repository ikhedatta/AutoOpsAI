## Key Requirements

**agent/store/database.py:**

```python
from motor.motor_asyncio import AsyncIOMotorClient
from beanie import init_beanie
from agent.config import get_settings
from agent.store.documents import IncidentDocument, UserDocument, AuditLogDocument, ApprovalDocument

_motor_client: AsyncIOMotorClient | None = None

async def init_db() -> None:
    """
    Initialise Motor client and Beanie ODM.
    Called once at application startup from the FastAPI lifespan.
    """
    global _motor_client
    settings = get_settings()
    _motor_client = AsyncIOMotorClient(settings.mongodb_url)
    db = _motor_client[settings.mongodb_database]
    await init_beanie(
        database=db,
        document_models=[IncidentDocument, UserDocument, AuditLogDocument, ApprovalDocument],
    )

async def close_db() -> None:
    """Close Motor client. Called from lifespan on shutdown."""
    global _motor_client
    if _motor_client:
        _motor_client.close()
        _motor_client = None

async def check_db_health() -> bool:
    """Ping MongoDB. Returns True if reachable."""
    if not _motor_client:
        return False
    try:
        await _motor_client.admin.command("ping")
        return True
    except Exception:
        return False

def get_motor_client() -> AsyncIOMotorClient:
    if not _motor_client:
        raise RuntimeError("Database not initialised. Call init_db() first.")
    return _motor_client
```

**agent/store/documents.py:**

All documents use Beanie's `Document` base class. Use `from beanie import Document, Indexed` for indexed fields. Use `PydanticObjectId` for IDs.

```python
class TimelineEvent(BaseModel):
    """Embedded in IncidentDocument. Each event is one step in the incident lifecycle."""
    timestamp: datetime
    event_type: str          # "detected", "approval_requested", "approved", "denied",
                             # "remediating", "step_complete", "step_failed",
                             # "verified", "resolved", "escalated", "rolled_back"
    actor: Optional[str] = None   # username or "agent" or "system"
    detail: str
    metadata: dict = {}


class IncidentDocument(Document):
    incident_id: str                    # e.g. "inc_20260401_001" — generated by engine
    title: str
    severity: str                       # RiskLevel value
    status: str                         # IncidentStatus value
    service: str
    playbook_id: Optional[str] = None
    detected_at: datetime
    resolved_at: Optional[datetime] = None
    resolution_time_seconds: Optional[float] = None
    auto_resolved: bool = False
    diagnosis_summary: Optional[str] = None
    diagnosis_confidence: Optional[float] = None
    llm_reasoning: Optional[str] = None
    proposed_action: Optional[str] = None
    rollback_plan: Optional[str] = None
    metrics_at_detection: dict = {}
    timeline: list[TimelineEvent] = []  # embedded subdocument list

    class Settings:
        name = "incidents"
        indexes = [
            "status",
            "severity",
            "service",
            "detected_at",
            [("status", 1), ("detected_at", -1)],   # compound
        ]


class UserDocument(Document):
    username: Indexed(str, unique=True)
    email: Optional[str] = None
    hashed_password: str
    role: str                    # "admin", "operator", "viewer"
    teams_upn: Optional[str] = None   # Azure AD UPN for Teams user mapping
    is_active: bool = True
    created_at: datetime
    last_login: Optional[datetime] = None

    class Settings:
        name = "users"


class AuditLogDocument(Document):
    timestamp: datetime
    user_id: Optional[str] = None     # PydanticObjectId as str; None for system actions
    username: Optional[str] = None
    action: str                        # e.g. "approve_incident", "edit_playbook", "login"
    resource_type: str                 # "incident", "playbook", "user", "agent_config"
    resource_id: Optional[str] = None
    detail: str
    ip_address: Optional[str] = None
    result: str = "success"            # "success" or "failure"

    class Settings:
        name = "audit_log"
        indexes = [
            "timestamp",
            "user_id",
            [("timestamp", -1)],
        ]


class ApprovalDocument(Document):
    incident_id: str
    status: str                         # ApprovalStatus value
    risk_level: str                     # RiskLevel value
    proposed_action: str
    rollback_plan: Optional[str] = None
    requested_at: datetime
    timeout_at: Optional[datetime] = None    # None for HIGH risk (no timeout)
    decided_at: Optional[datetime] = None
    decided_by: Optional[str] = None
    reason: Optional[str] = None
    teams_activity_id: Optional[str] = None          # for updating Teams card
    teams_conversation_reference: Optional[dict] = None  # for proactive messaging

    class Settings:
        name = "approvals"
        indexes = [
            "incident_id",
            "status",
            [("status", 1), ("requested_at", -1)],
        ]
```

**Important:** Beanie Documents are completely separate from the Pydantic models in `agent/models.py`. The Pydantic models are used for in-memory data flow between components. The Beanie Documents are for persistence only. Do NOT make `IncidentDocument` extend `agent.models` classes.

**TTL index for AuditLogDocument:** Add a TTL index on `timestamp` with expiry of 90 days. Beanie supports TTL indexes via `pymongo.IndexModel`:
```python
from pymongo import IndexModel, ASCENDING
import pymongo

class Settings:
    name = "audit_log"
    indexes = [
        IndexModel([("timestamp", ASCENDING)], expireAfterSeconds=90 * 24 * 3600),
        IndexModel([("user_id", ASCENDING)]),
    ]
```
